133function! s:SID_PREFIX()
134  return matchstr(expand('<sfile>'), '<SNR>\d\+_\zeSID_PREFIX$')
135endfunction

137" Set tabline.
138function! s:my_tabline()  "{{{
139  let s = ''
140  for i in range(1, tabpagenr('$'))
141    let bufnrs = tabpagebuflist(i)
142    let bufnr = bufnrs[tabpagewinnr(i) - 1]  " first window, first appears
143    let no = i  " display 0-origin tabpagenr.
144    let mod = getbufvar(bufnr, '&modified') ? '!' : ' '
145    let title = fnamemodify(bufname(bufnr), ':t')
146    let title = '[' . title . ']'
147    let s .= '%'.i.'T'
148    let s .= '%#' . (i == tabpagenr() ? 'TabLineSel' : 'TabLine') . '#'
149    let s .= no . ':' . title
150    let s .= mod
151    let s .= '%#TabLineFill# '
152  endfor
153  let s .= '%#TabLineFill#%T%=%#TabLine#'
154  return s
155endfunction "}}}
